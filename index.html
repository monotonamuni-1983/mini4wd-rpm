let maxRpm = 0; // ★ 最大RPM記録 function drawRpmFlow(rpm) { const w = rpmFlowCanvas.width; const h = rpmFlowCanvas.height; // 60秒分のデータ保持（300点） rpmHistory.push(rpm); if (rpmHistory.length > 300) rpmHistory.shift(); // ★ 最大RPM更新 if (rpm > maxRpm) maxRpm = rpm; rpmFlowCtx.clearRect(0, 0, w, h); // =============================== // グリッド線（RPM基準） // =============================== rpmFlowCtx.strokeStyle = "#e0e0e0"; rpmFlowCtx.lineWidth = 1; const rpmGrid = [10000, 20000, 30000, 40000, 50000, 60000]; rpmGrid.forEach(r => { const y = h - ((r - 5000) / (60000 - 5000)) * h; rpmFlowCtx.beginPath(); rpmFlowCtx.moveTo(0, y); rpmFlowCtx.lineTo(w, y); rpmFlowCtx.stroke(); rpmFlowCtx.fillStyle = "#666"; rpmFlowCtx.font = "12px sans-serif"; rpmFlowCtx.fillText(`${r} RPM`, 5, y - 2); }); // =============================== // 最大RPMライン（赤） // =============================== if (maxRpm > 5000) { const maxY = h - ((maxRpm - 5000) / (60000 - 5000)) * h; rpmFlowCtx.strokeStyle = "#ff0000"; rpmFlowCtx.lineWidth = 2; rpmFlowCtx.beginPath(); rpmFlowCtx.moveTo(0, maxY); rpmFlowCtx.lineTo(w, maxY); rpmFlowCtx.stroke(); rpmFlowCtx.fillStyle = "#ff0000"; rpmFlowCtx.font = "12px sans-serif"; rpmFlowCtx.fillText(`MAX ${maxRpm.toFixed(0)} RPM`, 5, maxY - 4); } // =============================== // 縦方向グリッド（時間軸） // =============================== for (let x = w; x > 0; x -= w / 12) { // 5秒ごと rpmFlowCtx.beginPath(); rpmFlowCtx.moveTo(x, 0); rpmFlowCtx.lineTo(x, h); rpmFlowCtx.stroke(); const sec = Math.round((w - x) / w * 60); rpmFlowCtx.fillStyle = "#666"; rpmFlowCtx.fillText(`${sec}s`, x + 2, 12); } // =============================== // RPMライン描画（右→左へ流れる） // =============================== rpmFlowCtx.strokeStyle = "#0077ff"; rpmFlowCtx.lineWidth = 2; rpmFlowCtx.beginPath(); for (let i = 0; i < rpmHistory.length; i++) { const x = w - (i * (w / 300)); // 300点＝60秒 const y = h - ((rpmHistory[i] - 5000) / (60000 - 5000)) * h; if (i === 0) rpmFlowCtx.moveTo(x, y); else rpmFlowCtx.lineTo(x, y); } rpmFlowCtx.stroke(); }
